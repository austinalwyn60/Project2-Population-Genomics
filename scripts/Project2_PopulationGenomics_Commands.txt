# Create shell script for variant extraction
cat > scripts/extract_variants.sh << 'EOF'
#!/bin/bash

# Script: Extract pharmacogenomic variants from 1000 Genomes VCF

INPUT_VCF="data/vcf_files/1000_genomes.vcf.gz"
VARIANT_LIST="data/raw_variants/variants_to_extract.txt"
OUTPUT_VCF="results/pharm_variants.vcf.gz"

echo "Extracting variants from VCF file..."

# Extract multiple variants using bcftools
bcftools view \
  -i 'ID=@'$VARIANT_LIST'' \
  $INPUT_VCF \
  -Oz \
  -o $OUTPUT_VCF

echo "Variant extraction complete: $OUTPUT_VCF"

# Index the extracted VCF
bcftools index $OUTPUT_VCF

echo "VCF file indexed successfully"
EOF

# Make script executable
chmod +x scripts/extract_variants.sh

# Create shell script for frequency calculation
cat > scripts/calculate_frequencies.sh << 'EOF'
#!/bin/bash

# Script: Calculate population-specific allele frequencies

INPUT_VCF="results/pharm_variants.vcf.gz"
OUTPUT_FILE="results/frequencies.txt"

echo "Calculating allele frequencies..."

# Add allele frequency information to VCF
bcftools +fill-tags $INPUT_VCF -- -t AF > results/pharm_variants_AF.vcf.gz

# Index the new VCF
bcftools index results/pharm_variants_AF.vcf.gz

# Query and export results
# Format: Variant ID, Allele Frequency
bcftools query \
  -f '%ID\t%AF\n' \
  results/pharm_variants_AF.vcf.gz \
  > $OUTPUT_FILE

echo "Allele frequencies calculated and saved to: $OUTPUT_FILE"

# Display results
echo "Results preview:"
cat $OUTPUT_FILE
EOF

# Make script executable
chmod +x scripts/calculate_frequencies.sh

# Run the frequency calculation
./scripts/calculate_frequencies.sh

# View the results
cat results/frequencies.txt

  -S <(grep "EAS" results/samples.txt) \
  results/pharm_variants.vcf.gz \
  -Oz \
  -o results/pharm_variants_EAS.vcf.gz

# Extract African (AFR) samples

  -S <(grep "AFR" results/samples.txt) \
  results/pharm_variants.vcf.gz \
  -Oz \
  -o results/pharm_variants_AFR.vcf.gz

# Calculate frequencies for each population
for POP in SAS EUR EAS AFR; do
  bcftools query \
    -f '%ID\t%AF\n' \
    results/pharm_variants_${POP}.vcf.gz \
    > results/frequencies_${POP}.txt
done

echo "Population-specific frequency files created"

# Create R analysis script
cat > scripts/analysis.R << 'EOF'
#!/usr/bin/env Rscript

# R Script: Pharmacogenomic Variant Frequency Analysis
# Install required packages if not already installed

if (!require("ggplot2")) {
  install.packages("ggplot2")
}

if (!require("reshape2")) {
  install.packages("reshape2")
}

library(ggplot2)
library(reshape2)

# Set working directory
setwd(".")

# Step 1: Create variant frequency table
# Replace values with your actual calculated frequencies
freq <- data.frame(
  Variant = c("rs4244285", "rs3892097", "rs1142345", "rs4149056"),
  SAS = c(0.32, 0.25, 0.08, 0.15),
  EUR = c(0.15, 0.10, 0.02, 0.08),
  AFR = c(0.10, 0.12, 0.01, 0.05),
  EAS = c(0.28, 0.30, 0.05, 0.12)
)

# Step 2: Save frequency table
write.table(freq, "results/allele_frequency_table.txt", 
            sep="\t", row.names=FALSE, quote=FALSE)

print("Frequency table:")
print(freq)

# Step 3: Convert to long format for ggplot2
freq_long <- melt(freq, id.vars = "Variant")

# Step 4: Rename columns for clarity
colnames(freq_long) <- c("Variant", "Population", "Allele_Frequency")

# Step 5: Create comparison plot
plot <- ggplot(freq_long, 
               aes(x = Variant, 
                   y = Allele_Frequency, 
                   fill = Population)) +
  geom_bar(stat = "identity", position = "dodge") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1),
        legend.position = "right") +
  labs(title = "Pharmacogenomic Variant Allele Frequencies Across Populations",
       x = "Variant (rsID)",
       y = "Allele Frequency",
       fill = "Population") +
  scale_fill_manual(values = c("SAS" = "#FF6B6B", 
                               "EUR" = "#4ECDC4", 
                               "AFR" = "#45B7D1", 
                               "EAS" = "#FFA07A"))

# Step 6: Save the plot
ggsave("plots/Population_Frequency_Comparison.png", 
       plot, width = 10, height = 6, dpi = 300)

print("Plot saved to plots/Population_Frequency_Comparison.png")

# Step 7: Generate summary statistics
summary_stats <- data.frame(
  Variant = freq$Variant,
  Max_Frequency = apply(freq[, -1], 1, max),
  Min_Frequency = apply(freq[, -1], 1, min),
  Range = apply(freq[, -1], 1, max) - apply(freq[, -1], 1, min),
  SAS_vs_EUR_Diff = freq$SAS - freq$EUR
)

write.table(summary_stats, "results/summary_statistics.txt",
            sep = "\t", row.names = FALSE, quote = FALSE)

print("Summary Statistics:")
print(summary_stats)

# Step 8: Identify largest differences
largest_diff <- summary_stats[order(summary_stats$SAS_vs_EUR_Diff, decreasing = TRUE), ]
print("Variants with largest SAS vs EUR differences:")
print(largest_diff)

print("R analysis complete!")
EOF

# Create validation script
cat > scripts/validate_gnomad.R << 'EOF'
#!/usr/bin/env Rscript

# Script: Validate frequencies against gnomAD data

# Read your calculated frequencies
freq_1000g <- read.table("results/allele_frequency_table.txt", 
                         header = TRUE, sep = "\t")

# Create gnomAD comparison table
# Note: Download actual gnomAD data from https://gnomad.broadinstitute.org/
# For this example, using hypothetical gnomAD values

gnomad_sas <- data.frame(
  Variant = c("rs4244285", "rs3892097", "rs1142345", "rs4149056"),
  gnomAD_SAS_AF = c(0.31, 0.24, 0.08, 0.15),
  Source = "gnomAD"
)

# Merge and compare
comparison <- merge(freq_1000g[, c("Variant", "SAS")], 
                    gnomad_sas[, c("Variant", "gnomAD_SAS_AF")],
                    by = "Variant")

colnames(comparison) <- c("Variant", "1000G_SAS", "gnomAD_SAS")
comparison$Difference <- abs(comparison$`1000G_SAS` - comparison$gnomAD_SAS)
comparison$Percent_Diff <- (comparison$Difference / comparison$gnomAD_SAS) * 100

# Save comparison
write.table(comparison, "validation/gnomAD_comparison.txt",
            sep = "\t", row.names = FALSE, quote = FALSE)

print("Validation Summary:")
print(comparison)

print("Average difference with gnomAD:")
print(paste("Mean absolute difference:", 
            round(mean(comparison$Difference), 4)))
print(paste("Mean percent difference:", 
            round(mean(comparison$Percent_Diff), 2), "%"))
EOF

chmod +x scripts/validate_gnomad.R
